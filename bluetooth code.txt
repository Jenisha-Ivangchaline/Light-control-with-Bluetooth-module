##  Arduino Code

#define RELAY_PIN 8

void setup() {
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW);
  Serial.begin(9600);
}

void loop() {
  if (Serial.available()) {
    String command = Serial.readStringUntil('\n');
    command.trim();
    if (command == "ON") {
      digitalWrite(RELAY_PIN, HIGH);
    } else if (command == "OFF") {
      digitalWrite(RELAY_PIN, LOW);
    }
  }
}
